/* K1 Kart Log receiver shell v5 — fetches logic from the repo.
   The real code lives at google-sheet-sync/logic.gs in
   https://github.com/robbietay1008-dev/K1-Kart-log */

var LOGIC_URL = 'https://raw.githubusercontent.com/robbietay1008-dev/K1-Kart-log/main/google-sheet-sync/logic.gs';

function _code() {
  var cache = CacheService.getScriptCache();
  var c = cache.get('klogic');
  if (!c) {
    var r = UrlFetchApp.fetch(LOGIC_URL, { muteHttpExceptions: true });
    if (r.getResponseCode() !== 200) throw new Error('logic fetch failed: HTTP ' + r.getResponseCode());
    c = r.getContentText();
    cache.put('klogic', c, 600); // re-check GitHub every 10 minutes
  }
  return c;
}

function doPost(e) {
  try { eval(_code()); return handlePost(e); }
  catch (err) { return ContentService.createTextOutput('error: ' + err); }
}
function doGet(e) {
  try { eval(_code()); return handleGet(e); }
  catch (err) { return ContentService.createTextOutput('error: ' + err); }
}
function onOpen() {
  try { eval(_code()); buildMenu(); } catch (err) {}
}
function placeOrder() { eval(_code()); placeOrderImpl(); }
function bookReceived() { eval(_code()); bookReceivedImpl(); }
function cleanupSheet() { eval(_code()); cleanupImpl(); }
function refreshLogic() {
  CacheService.getScriptCache().remove('klogic');
  eval(_code());
  SpreadsheetApp.getUi ? void 0 : void 0;
}
